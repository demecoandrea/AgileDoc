// Algoritmo per stampare i numeri primi da 1 a 100
START
  FOR i da 1 a 100
    SE èPrimo(i) = VERO
      STAMPA(i)
    FINE SE
  FINE FOR
FINE

// Sottoprocedura per verificare se un numero è primo
FUNZIONE èPrimo(n)
  SE n < 2
    RITORNA FALSO
  FINE SE

  PER k da 2 a SQRT(n)
    SE n MOD k = 0
      RITORNA FALSO
    FINE SE
  FINE PER

  RITORNA VERO
FINE FUNZIONE
